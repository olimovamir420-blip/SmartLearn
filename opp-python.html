Ushbu kursda siz obyektga yo‘naltirilgan dasturlash (OOP) asoslarini o‘rganasiz:

class Talaba:
    def __init__(self, ism, yosh):
        self.ism = ism
        self.yosh = yosh

    def salom_ber(self):
        print(f"Salom, mening ismim {self.ism}, yoshim {self.yosh}da.")

student = Talaba("Amirbek", 20)
student.salom_ber()
