1. Sinf (Class) va obyekt (Object)

class Talaba:
    def __init__(self, ism, yosh):
        self.ism = ism
        self.yosh = yosh

    def tanishtir(self):
        print(f"Men {self.ism}, yoshim {self.yosh}")

ali = Talaba("Ali", 21)
ali.tanishtir()

2. Meros olish (Inheritance)

class Hayvon:
    def ovoz_ber(self):
        print("Hayvon ovozi")

class It(Hayvon):
    def ovoz_ber(self):
        print("Vov-vov")

dog = It()
dog.ovoz_ber()

3. Inkapsulyatsiya (Encapsulation)

class BankHisob:
    def __init__(self):
        self.__balans = 0

    def pul_qosh(self, miqdor):
        self.__balans += miqdor

    def balansni_korish(self):
        return self.__balans

hisob = BankHisob()
hisob.pul_qosh(100)
print(hisob.balansni_korish())

4. Polimorfizm (Polymorphism)

class Qush:
    def ovoz(self):
        print("Chir-chir")

class Toy:
    def ovoz(self):
        print("Gug-gug")

for hayvon in (Qush(), Toy()):
    hayvon.ovoz()
